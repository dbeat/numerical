142 /*
143  * For a sparse matrix, return a vector of triplets, such that we can
144  * reconstruct the matrix using setFromTriplet function.
145  * @param matrix A sparse matrix.
146  * @return A triplet with the row, column and value of the non-zero entries.
147  */
148 template <typename Derived>
149 std::vector<Eigen::Triplet<typename Derived::Scalar>> SparseMatrixToTriplets(
150  const Derived& matrix) {
151  using Scalar = typename Derived::Scalar;
152  std::vector<Eigen::Triplet<Scalar>> triplets;
153  triplets.reserve(matrix.nonZeros());
154  for (int i = 0; i < matrix.outerSize(); i++) {
155  for (typename Derived::InnerIterator it(matrix, i); it; ++it) {
156  triplets.push_back(
157  Eigen::Triplet<Scalar>(it.row(), it.col(), it.value()));
158  }
159  }
160  return triplets;
161 }
